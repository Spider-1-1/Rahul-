1. Program to copy all elements of one array into another array.

Code:       
     public class CopyArray {
                public static void main(String[] args) {
                    int[] original = {1, 2, 3, 4, 5};
                    int[] copy = new int[original.length];
            
                    System.arraycopy(original, 0, copy, 0, original.length);
            
                    System.out.println("Copied Array: ");
                    for (int num : copy) {
                        System.out.print(num + " ");
                    }
                }
            }
Output:
Copied Array: 1 2 3 4 5


2. Program to find the frequency of each element in the array.

Code:       
     import java.util.HashMap;
            
            public class FrequencyArray {
                public static void main(String[] args) {
                    int[] array = {1, 2, 2, 3, 3, 3, 4};
                    HashMap frequencyMap = new HashMap<>();
            
                    for (int num : array) {
                        frequencyMap.put(num, frequencyMap.getOrDefault(num, 0) + 1);
                    }
            
                    System.out.println("Frequency of elements: " + frequencyMap);
                }
            }
Output:
Frequency of elements: {1=1, 2=2, 3=3, 4=1}


3. Program to left rotate the elements of an array.

Code:      
      public class LeftRotateArray {
                public static void main(String[] args) {
                    int[] array = {1, 2, 3, 4, 5};
                    int n = array.length;
                    int temp = array[0];
            
                    for (int i = 0; i < n - 1; i++) {
                        array[i] = array[i + 1];
                    }
                    array[n - 1] = temp;
            
                    System.out.println("Left Rotated Array: ");
                    for (int num : array) {
                        System.out.print(num + " ");
                    }
                }
            }
Output:
Left Rotated Array: 2 3 4 5 1


4. Program to print the duplicate elements of an array.

Code:   
         import java.util.HashSet;
            
            public class DuplicateElements {
                public static void main(String[] args) {
                    int[] array = {1, 2, 3, 4, 4, 5, 1};
                    HashSet duplicates = new HashSet<>();
                    HashSet seen = new HashSet<>();
            
                    for (int num : array) {
                        if (!seen.add(num)) {
                            duplicates.add(num);
                        }
                    }
            
                    System.out.println("Duplicate Elements: " + duplicates);
                }
            }
Output:
Duplicate Elements: [1, 4]

13. Program to right rotate the elements of an array.

Code:     
       public class RightRotateArray {
                public static void main(String[] args) {
                    int[] array = {1, 2, 3, 4, 5};
                    int n = array.length;
                    int temp = array[n - 1];
            
                    for (int i = n - 1; i > 0; i--) {
                        array[i] = array[i - 1];
                    }
                    array[0] = temp;
            
                    System.out.println("Right Rotated Array: ");
                    for (int num : array) {
                        System.out.print(num + " ");
                    }
                }
            }
Output:
Right Rotated Array: 5 1 2 3 4

21. Program to remove duplicate elements in an array.

Code:   
         import java.util.Arrays;
            
            public class RemoveDuplicates {
                public static void main(String[] args) {
                    int[] array = {1, 2, 2, 3, 4, 4, 5};
                    int[] uniqueArray = Arrays.stream(array).distinct().toArray();
            
                    System.out.println("Array after removing duplicates: ");
                    for (int num : uniqueArray) {
                        System.out.print(num + " ");
                    }
                }
            }
Output:
Array after removing duplicates: 1 2 3 4 5


22. Program to print odd and even numbers from an array.

Code:       
     public class OddEvenArray {
                public static void main(String[] args) {
                    int[] array = {1, 2, 3, 4, 5};
            
                    System.out.println("Odd Numbers: ");
                    for (int num : array) {
                        if (num % 2 != 0) {
                            System.out.print(num + " ");
                        }
                    }
            
                    System.out.println("\nEven Numbers: ");
                    for (int num : array) {
                        if (num % 2 == 0) {
                            System.out.print(num + " ");
                        }
                    }
                }
            }
Output:
Odd Numbers: 1 3 5, Even Numbers: 2 4
